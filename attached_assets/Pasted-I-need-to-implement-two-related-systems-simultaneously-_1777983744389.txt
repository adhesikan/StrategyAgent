PLAN ENFORCEMENT MIDDLEWARE
CREATE: server/middleware/planGuard.ts

requireFeature(feature: string):
  - Check canAccessFeature(user.planId, feature)
  - If false: return 403 {
      error: 'PLAN_LIMIT',
      feature,
      currentPlan: planId,
      requiredPlan: getRequiredPlan(feature),
      message: 'Upgrade to access this feature'
    }

requireAnalysisQuota():
  - Call resetDailyAnalysesIfNeeded(userId)
  - Check limit (-1 = unlimited)
  - If exceeded: return 429 {
      error: 'DAILY_LIMIT',
      used, limit,
      resetsAt: tomorrow midnight UTC,
      currentPlan: planId,
      upgradePlan: getUpgradePlan(planId)
    }
  - If within limit: increment counter, pass through

Apply to existing routes:
  POST /api/trade-chat → requireAnalysisQuota()
  POST /api/trade-setup → requireAnalysisQuota()
  GET /api/scanner/* → requireFeature('scanner')
  POST /api/orders/* → requireFeature('brokerConnect')
  POST /api/automation/* → requireFeature('automation')
  GET /api/options-flow/* → requireFeature('optionsFlow')
